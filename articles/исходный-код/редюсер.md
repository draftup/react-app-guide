# Редюсер

Функция, которая создает новый объект заданного типа в ответ на переданную команду.

```ts

export const reduceToNextAppState = (state: AppState, event: AppEvent) => {
    switch (event.name) {
        case 'INC':
            return state + 1;
        case 'DEC':
            return state - 1;
        default:
            return state;
    }
}

```
