# Редюсер состояния

Функция, которая меняет объект состояния в ответ на событие.

```ts

export const appStateReducer = (state: AppState, event: AppEvent) => {
    switch (event.name) {
        case 'INC':
            return state + 1;
        case 'DEC':
            return state - 1;
        default:
            return state;
    }
}

```
